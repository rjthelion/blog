Date Wed 03 June 2020 Tags Pelican

Pelican is a static site generator

def do_twice(func):
def wrapper_do_twice(*args, **kwargs):
    return func(*args, **kwargs).lower()
return wrapper_do_twice

@do_twice
def say_whee(some_text):
    print(some_text)

x = 'Whee!'
say_whee(x)
